section ‹ Defining runs ›

abbreviation hamlet where "hamlet ≡ fst"
abbreviation time where "time ≡ snd"


print_classes

type_synonym 'τ instant = "clock ⇒ (bool × 'τ tag_const)"
typedef (overloaded) 'τ::linordered_field run =
  "{ ρ::nat ⇒ 'τ instant. ∀c. mono (λn. time (ρ n c)) }"
proof
  show "(λ_ _. (True, τ⇩c⇩s⇩t 0)) ∈ {ρ::nat ⇒ clock ⇒ bool × 'τ tag_const. ∀c::clock. mono (λn::nat. time (ρ n c))}"
  using mono_def by auto
qed
lemma Abs_run_inverse_rewrite:
  "∀c. mono (λn. time (ρ n c)) ⟹ Rep_run (Abs_run ρ) = ρ"
  by (simp add: Abs_run_inverse)

(* WARNING: Admitting monotonicity to compute faster. Use for debugging purposes only. *)
lemma Abs_run_inverse_rewrite_unsafe:
  "Rep_run (Abs_run ρ) = ρ"
  oops (* Use [sorry] when testing *)

(* Counts the number of ticks in run [ρ] on clock [K] from indexes 0 to [n] *)
fun run_tick_count :: "('τ::linordered_field) run ⇒ clock ⇒ nat ⇒ nat" ("#⇧≤ _ _ _") where
    "#⇧≤ ρ K 0       = (if hamlet ((Rep_run ρ) 0 K)
                       then 1
                       else 0)"
  | "#⇧≤ ρ K (Suc n) = (if hamlet ((Rep_run ρ) (Suc n) K)
                       then 1 + #⇧≤ ρ K n
                       else #⇧≤ ρ K n)"

(* Counts the number of ticks in run [ρ] on clock [K] from indexes 0 to [n - 1] *)
fun run_tick_count_strictly :: "('τ::linordered_field) run ⇒ clock ⇒ nat ⇒ nat" ("#⇧< _ _ _") where
    "#⇧< ρ K 0       = 0"
  | "#⇧< ρ K (Suc n) = #⇧≤ ρ K n"

fun counter_expr_eval :: "('τ::linordered_field) run ⇒ cnt_expr ⇒ nat" ("⟦ _ ⊢ _ ⟧⇩c⇩n⇩t⇩e⇩x⇩p⇩r") where
    "⟦ ρ ⊢ #⇧< clk indx ⟧⇩c⇩n⇩t⇩e⇩x⇩p⇩r = run_tick_count_strictly ρ clk indx"
  | "⟦ ρ ⊢ #⇧≤ clk indx ⟧⇩c⇩n⇩t⇩e⇩x⇩p⇩r = run_tick_count ρ clk indx"

fun symbolic_run_interpretation_primitive :: "('τ::linordered_field) constr ⇒ 'τ run set" ("⟦ _ ⟧⇩p⇩r⇩i⇩m") where
    "⟦ K ⇑ n  ⟧⇩p⇩r⇩i⇩m   = { ρ. hamlet ((Rep_run ρ) n K) = True }"
  | "⟦ K ¬⇑ n ⟧⇩p⇩r⇩i⇩m   = { ρ. hamlet ((Rep_run ρ) n K) = False }"
  | "⟦ K ¬⇑ < n ⟧⇩p⇩r⇩i⇩m   = { ρ. ∀i < n. hamlet ((Rep_run ρ) i K) = False }"
  | "⟦ K ¬⇑ ≥ n ⟧⇩p⇩r⇩i⇩m   = { ρ. ∀i ≥ n. hamlet ((Rep_run ρ) i K) = False }"
  | "⟦ K ⇓ n @ ⦇ τ ⦈ ⟧⇩p⇩r⇩i⇩m = { ρ. time ((Rep_run ρ) n K) = τ }"
  | "⟦ K ⇓ n @ ⦇ τ⇩v⇩a⇩r(K', n') ⊕ τ ⦈ ⟧⇩p⇩r⇩i⇩m = { ρ. time ((Rep_run ρ) n K) = time ((Rep_run ρ) n' K') + τ }"
  | "⟦ ⌊τ⇩v⇩a⇩r(K⇩1, n⇩1), τ⇩v⇩a⇩r(K⇩2, n⇩2)⌋ ∈ R ⟧⇩p⇩r⇩i⇩m = { ρ. R (time ((Rep_run ρ) n⇩1 K⇩1), time ((Rep_run ρ) n⇩2 K⇩2)) }"
  | "⟦ ⌈e⇩1, e⇩2⌉ ∈ R ⟧⇩p⇩r⇩i⇩m = { ρ. R (⟦ ρ ⊢ e⇩1 ⟧⇩c⇩n⇩t⇩e⇩x⇩p⇩r, ⟦ ρ ⊢ e⇩2 ⟧⇩c⇩n⇩t⇩e⇩x⇩p⇩r) }"
  | "⟦ cnt_e⇩1 ≼ cnt_e⇩2 ⟧⇩p⇩r⇩i⇩m = { ρ. ⟦ ρ ⊢ cnt_e⇩1 ⟧⇩c⇩n⇩t⇩e⇩x⇩p⇩r ≤ ⟦ ρ ⊢ cnt_e⇩2 ⟧⇩c⇩n⇩t⇩e⇩x⇩p⇩r }"
  (* | "⟦ (τ⇩v⇩a⇩r(K⇩1, n⇩1)) ≐ α * τ⇩v⇩a⇩r(K⇩2, n⇩2) + β ⟧⇩p⇩r⇩i⇩m = { ρ. time ((Rep_run ρ) n⇩1 K⇩1) = α * time ((Rep_run ρ) n⇩2 K⇩2) + β }" *)

fun symbolic_run_interpretation :: "('τ::linordered_field) constr list ⇒ ('τ::linordered_field) run set" ("⟦⟦ _ ⟧⟧⇩p⇩r⇩i⇩m") where
    "⟦⟦ [] ⟧⟧⇩p⇩r⇩i⇩m = { _. True }"
  | "⟦⟦ γ # Γ ⟧⟧⇩p⇩r⇩i⇩m = ⟦ γ ⟧⇩p⇩r⇩i⇩m ∩ ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m"

lemma symbolic_run_interp_cons_morph:
  "⟦ γ ⟧⇩p⇩r⇩i⇩m ∩ ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ γ # Γ ⟧⟧⇩p⇩r⇩i⇩m"
  by auto

definition consistent_run :: "('τ::linordered_field) constr list ⇒ bool" where 
  "consistent_run Γ ≡ ∃ρ. ρ ∈ ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m"

(**) section ‹Defining a method for witness construction› (**)

(* Initial states *)
abbreviation initial_run :: "('τ::linordered_field) run" ("ρ⇩⊙") where
  "ρ⇩⊙ ≡ Abs_run ((λ_ _. (False, τ⇩c⇩s⇩t 0)) ::nat ⇒ clock ⇒ (bool × 'τ tag_const))"

(* To ensure monotonicity, time tag is set at a specific instant and forever after (stuttering) *)
fun time_update
  :: "nat ⇒ clock ⇒ ('τ::linordered_field) tag_const ⇒ (nat ⇒ clock ⇒ (bool × 'τ tag_const)) ⇒ (nat ⇒ clock ⇒ (bool × 'τ tag_const))" where
    "time_update n K τ ρ = (λn' K'. if K = K' ∧ n ≤ n' then (hamlet (ρ n K), τ) else ρ n' K')"


(**) section ‹Rules and properties of consistence› (**)

(* declare [[show_sorts]] *)

lemma context_consistency_preservationI:
  "consistent_run ((γ :: ('τ::linordered_field) constr) # Γ) ⟹ consistent_run Γ"
unfolding consistent_run_def
by auto

(* This is very restrictive *)
inductive context_independency :: "('τ::linordered_field) constr ⇒ 'τ constr list ⇒ bool" ("_ ⨝ _") where
  NotTicks_independency:
  "(K ⇑ n) ∉ set Γ ⟹ (K ¬⇑ n) ⨝ Γ"
| Ticks_independency:
  "(K ¬⇑ n) ∉ set Γ ⟹ (K ⇑ n) ⨝ Γ"
| Timestamp_independency:
  "(∄τ'. τ' = τ ∧ (K ⇓ n @ τ) ∈ set Γ) ⟹ (K ⇓ n @ τ) ⨝ Γ"

thm context_independency.induct

lemma context_consistency_preservationE:
  assumes consist: "consistent_run Γ"
  and     indepen: "γ ⨝ Γ"
  shows "consistent_run (γ # Γ)"
  oops

(**) section ‹Fixpoint lemma› (**)

theorem symrun_interp_fixpoint:
  "⋂ ((λγ. ⟦ γ ⟧⇩p⇩r⇩i⇩m) ` set Γ) = ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m"
  proof (induct Γ)
    case Nil
    then show ?case by simp
  next
    case (Cons a Γ)
    then show ?case by auto
  qed

(**) section ‹Expansion law› (**)
text ‹Similar to the expansion laws of lattices›

theorem symrun_interp_expansion:
  shows "⟦⟦ Γ⇩1 @ Γ⇩2 ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ⇩1 ⟧⟧⇩p⇩r⇩i⇩m ∩ ⟦⟦ Γ⇩2 ⟧⟧⇩p⇩r⇩i⇩m"
  proof (induct Γ⇩1)
    case Nil
    then show ?case by simp
  next
    case (Cons a Γ⇩1)
    then show ?case by auto
  qed

(**) section ‹Equational laws for TESL formulae denotationally interpreted› (**)
(***) subsection ‹General laws› (***)

lemma symrun_interp_assoc:
  shows "⟦⟦ (Γ⇩1 @ Γ⇩2) @ Γ⇩3 ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ⇩1 @ (Γ⇩2 @ Γ⇩3) ⟧⟧⇩p⇩r⇩i⇩m"
  by auto

lemma symrun_interp_commute:
  shows "⟦⟦ Γ⇩1 @ Γ⇩2 ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ⇩2 @ Γ⇩1 ⟧⟧⇩p⇩r⇩i⇩m"
  by (simp add: symrun_interp_expansion inf_sup_aci(1))

lemma symrun_interp_left_commute:
  shows "⟦⟦ Γ⇩1 @ (Γ⇩2 @ Γ⇩3) ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ⇩2 @ (Γ⇩1 @ Γ⇩3) ⟧⟧⇩p⇩r⇩i⇩m"
  unfolding symrun_interp_expansion by auto

lemma symrun_interp_idem:
  shows "⟦⟦ Γ @ Γ ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m"
  using symrun_interp_expansion by auto

lemma symrun_interp_left_idem:
  shows "⟦⟦ Γ⇩1 @ (Γ⇩1 @ Γ⇩2) ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ⇩1 @ Γ⇩2 ⟧⟧⇩p⇩r⇩i⇩m"
  using symrun_interp_expansion by auto

lemma symrun_interp_right_idem:
  shows "⟦⟦ (Γ⇩1 @ Γ⇩2) @ Γ⇩2 ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ⇩1 @ Γ⇩2 ⟧⟧⇩p⇩r⇩i⇩m"
  unfolding symrun_interp_expansion by auto

lemmas symrun_interp_aci = symrun_interp_commute symrun_interp_assoc symrun_interp_left_commute symrun_interp_left_idem

(* Identity element *)
lemma symrun_interp_neutral1:
  shows "⟦⟦ [] @ Γ ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m"
  by simp

lemma symrun_interp_neutral2:
  shows "⟦⟦ Γ @ [] ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m"
  by simp

section ‹Decreasing interpretation of TESL formulae›

lemma TESL_sem_decreases_head:
  "⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m ⊇ ⟦⟦ γ # Γ ⟧⟧⇩p⇩r⇩i⇩m"
  by simp

lemma TESL_sem_decreases_tail:
  "⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m ⊇ ⟦⟦ Γ @ [γ] ⟧⟧⇩p⇩r⇩i⇩m"
  by (simp add: symrun_interp_expansion)

lemma symrun_interp_formula_stuttering:
  assumes bel: "γ ∈ set Γ"
  shows "⟦⟦ γ # Γ ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m"
  by (metis Int_absorb1 Int_left_commute bel inf_le1 split_list symbolic_run_interpretation.simps(2) symrun_interp_expansion)

lemma symrun_interp_decreases:
  shows "⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m ⊇ ⟦⟦ γ # Γ ⟧⟧⇩p⇩r⇩i⇩m"
  by (rule TESL_sem_decreases_head)

lemma symrun_interp_remdups_absorb:
  shows "⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ remdups Γ ⟧⟧⇩p⇩r⇩i⇩m"
  proof (induct Γ)
    case Nil
    then show ?case by simp
  next
    case (Cons a Γ)
    then show ?case
      using symrun_interp_formula_stuttering by auto
  qed

lemma symrun_interp_set_lifting:
  assumes "set Γ = set Γ'"
  shows "⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ' ⟧⟧⇩p⇩r⇩i⇩m"
  proof -     
    have "set (remdups Γ) = set (remdups Γ')"
      by (simp add: assms)
    moreover have fxpntΓ: "⋂ ((λγ. ⟦ γ ⟧⇩p⇩r⇩i⇩m) ` set Γ) = ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m"
      by (simp add: symrun_interp_fixpoint)
    moreover have fxpntΓ': "⋂ ((λγ. ⟦ γ ⟧⇩p⇩r⇩i⇩m) ` set Γ') = ⟦⟦ Γ' ⟧⟧⇩p⇩r⇩i⇩m"
      by (simp add: symrun_interp_fixpoint)
    moreover have "⋂ ((λγ. ⟦ γ ⟧⇩p⇩r⇩i⇩m) ` set Γ) = ⋂ ((λγ. ⟦ γ ⟧⇩p⇩r⇩i⇩m) ` set Γ')"
      by (simp add: assms)
    ultimately show ?thesis using symrun_interp_remdups_absorb by auto
  qed

theorem symrun_interp_decreases_setinc:
  assumes incl: "set Γ ⊆ set Γ'"
  shows "⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m ⊇ ⟦⟦ Γ' ⟧⟧⇩p⇩r⇩i⇩m"
  proof -
    obtain Γ⇩r where decompose: "set (Γ @ Γ⇩r) = set Γ'" using incl by auto
    have "set (Γ @ Γ⇩r) = set Γ'" using incl decompose by blast
    moreover have "(set Γ) ∪ (set Γ⇩r) = set Γ'" using incl decompose by auto
    moreover have "⟦⟦ Γ' ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ @ Γ⇩r ⟧⟧⇩p⇩r⇩i⇩m" using symrun_interp_set_lifting decompose by blast
    moreover have "⟦⟦ Γ @ Γ⇩r ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m ∩ ⟦⟦ Γ⇩r ⟧⟧⇩p⇩r⇩i⇩m" by (simp add: symrun_interp_expansion)
    moreover have "⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m ⊇ ⟦⟦ Γ ⟧⟧⇩p⇩r⇩i⇩m ∩ ⟦⟦ Γ⇩r ⟧⟧⇩p⇩r⇩i⇩m" by simp
    ultimately show ?thesis by simp
  qed

lemma symrun_interp_decreases_add_head:
  assumes incl: "set Γ ⊆ set Γ'"
  shows "⟦⟦ γ # Γ ⟧⟧⇩p⇩r⇩i⇩m ⊇ ⟦⟦ γ # Γ' ⟧⟧⇩p⇩r⇩i⇩m"
  using symrun_interp_decreases_setinc incl by auto

lemma symrun_interp_decreases_add_tail:
  assumes incl: "set Γ ⊆ set Γ'"
  shows "⟦⟦ Γ @ [γ] ⟧⟧⇩p⇩r⇩i⇩m ⊇ ⟦⟦ Γ' @ [γ] ⟧⟧⇩p⇩r⇩i⇩m"
  by (metis symrun_interp_commute symrun_interp_decreases_add_head append_Cons append_Nil incl)

lemma symrun_interp_absorb1:
  assumes incl: "set Γ⇩1 ⊆ set Γ⇩2"
  shows "⟦⟦ Γ⇩1 @ Γ⇩2 ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ⇩2 ⟧⟧⇩p⇩r⇩i⇩m"
  by (simp add: Int_absorb1 symrun_interp_decreases_setinc symrun_interp_expansion incl)

lemma symrun_interp_absorb2:
  assumes incl: "set Γ⇩2 ⊆ set Γ⇩1"
  shows "⟦⟦ Γ⇩1 @ Γ⇩2 ⟧⟧⇩p⇩r⇩i⇩m = ⟦⟦ Γ⇩1 ⟧⟧⇩p⇩r⇩i⇩m"
  using symrun_interp_absorb1 symrun_interp_commute incl by blast
