section ‹Stepwise denotational interpretation of TESL atoms›

(* Denotational interpretation of TESL bounded by index *)
fun TESL_interpretation_atomic_stepwise
    :: "TESL_atomic ⇒ nat ⇒ run set" ("⟦ _ ⟧⇩T⇩E⇩S⇩L⇗≥ _⇖") where
    "⟦ K sporadic τ ⟧⇩T⇩E⇩S⇩L⇗≥ i⇖ =
        { ρ. ∃n≥i. hamlet ((Rep_run ρ) n K) = True ∧ time ((Rep_run ρ) n K) = τ }"
  | "⟦ K⇩1 sporadic ⌊τ⌋ on K⇩2 ⟧⇩T⇩E⇩S⇩L⇗≥ i⇖ =
        { ρ. ∃n≥i. hamlet ((Rep_run ρ) n K⇩1) = True ∧ time ((Rep_run ρ) n K⇩2) = τ }"
  | "⟦ K⇩1 sporadic ⌊τ⇩v⇩a⇩r(K⇩p⇩a⇩s⇩t, n⇩p⇩a⇩s⇩t) ⊕ δτ⌋ on K⇩2 ⟧⇩T⇩E⇩S⇩L⇗≥ i⇖ =
        { ρ. ∃n≥i. hamlet ((Rep_run ρ) n K⇩1) = True
                         ∧ time ((Rep_run ρ) n K⇩2) = time ((Rep_run ρ) n⇩p⇩a⇩s⇩t K⇩p⇩a⇩s⇩t) + δτ }"
  | "⟦ tag-relation K⇩1 = α * K⇩2 + β ⟧⇩T⇩E⇩S⇩L⇗≥ i⇖ =
        { ρ. ∀n≥i. time ((Rep_run ρ) n K⇩1) = α * time ((Rep_run ρ) n K⇩2) + β }"
  | "⟦ tag-relation ⟨K⇩1, K⇩2⟩ ∈ R ⟧⇩T⇩E⇩S⇩L⇗≥ i⇖ =
        { ρ. ∀n≥i. R (time ((Rep_run ρ) n K⇩1), time ((Rep_run ρ) n K⇩2)) }"
  | "⟦ master implies slave ⟧⇩T⇩E⇩S⇩L⇗≥ i⇖ =
        { ρ. ∀n≥i. hamlet ((Rep_run ρ) n master) ⟶ hamlet ((Rep_run ρ) n slave) }"
  | "⟦ master time-delayed by δτ on measuring implies slave ⟧⇩T⇩E⇩S⇩L⇗≥ i⇖ =
        { ρ. ∀n≥i. hamlet ((Rep_run ρ) n master) ⟶
                 (let measured_time = time ((Rep_run ρ) n measuring) in
                  ∃m ≥ n. hamlet ((Rep_run ρ) m slave)
                          ∧ time ((Rep_run ρ) m measuring) = measured_time + δτ
                 )
        }"

theorem TESL_interp_unfold_stepwise_positive_atoms:
  assumes "positive_atom φ"
  shows "⟦ φ ⟧⇩T⇩E⇩S⇩L = ⋃ {Y. ∃n::nat. Y = ⟦ φ ⟧⇩T⇩E⇩S⇩L⇗≥ n⇖}"
  proof -
    obtain cc :: "TESL_atomic ⇒ clock" and tt :: "TESL_atomic ⇒ tag_const" and cca :: "TESL_atomic ⇒ clock" and tta :: "TESL_atomic ⇒ tag_expr" and ccb :: "TESL_atomic ⇒ clock" where
      f1: "∀t. ¬ positive_atom t ∨ t = cc t sporadic tt t ∨ t = cca t sporadic tta t on ccb t"
      using positive_atom.elims(2) by moura
    obtain ttb :: "tag_expr ⇒ tag_const" where
      "⋀c t ca. ⋃{R. ∃n. R = ⟦ c sporadic t on ca ⟧⇩T⇩E⇩S⇩L⇗≥ n⇖} = ⟦ c sporadic t on ca ⟧⇩T⇩E⇩S⇩L ∨ ⌊ ttb t ⌋ = t"
      by (metis TESL_interp_unfold_stepwise_sporadicon_add old.prod.exhaust tag_expr.exhaust tag_var.exhaust)
    then have "⋀t. ⋃{R. ∃n. R = ⟦ t ⟧⇩T⇩E⇩S⇩L⇗≥ n⇖} = ⟦ t ⟧⇩T⇩E⇩S⇩L ∨ ⌊ ttb (tta t) ⌋ = tta t ∨ ¬ positive_atom t"
      using f1 by (metis TESL_interp_unfold_stepwise_sporadic)
    then show ?thesis
      using f1 by (metis TESL_interp_unfold_stepwise_sporadic TESL_interp_unfold_stepwise_sporadicon assms)
  qed

theorem TESL_interp_unfold_stepwise_negative_atoms:
  assumes "¬ positive_atom φ"
  shows "⟦ φ ⟧⇩T⇩E⇩S⇩L = ⋂ {Y. ∃n::nat. Y = ⟦ φ ⟧⇩T⇩E⇩S⇩L⇗≥ n⇖}"
  by (smt Collect_cong TESL_interp_unfold_stepwise_implies TESL_interp_unfold_stepwise_tagrel TESL_interp_unfold_stepwise_tagrelgen TESL_interp_unfold_stepwise_timedelayed assms positive_atom.elims(3))

lemma TESL_interp_stepwise_sporadic_coind_unfold:
  shows "⟦ K sporadic τ ⟧⇩T⇩E⇩S⇩L⇗≥ n⇖ =
    ⟦ K ⇑ n ⟧⇩s⇩y⇩m⇩r⇩u⇩n ∩ ⟦ K ⇓ n @ ⌊ τ ⌋ ⟧⇩s⇩y⇩m⇩r⇩u⇩n
    ∪ ⟦ K sporadic τ ⟧⇩T⇩E⇩S⇩L⇗≥ Suc n⇖"
  proof -
    have "{ ρ. ∃m≥n. hamlet ((Rep_run ρ) m K) = True ∧ time ((Rep_run ρ) m K) = τ }
        = { ρ. hamlet ((Rep_run ρ) n K) = True ∧ time ((Rep_run ρ) n K) = τ
               ∨ (∃m≥Suc n. hamlet ((Rep_run ρ) m K) = True ∧ time ((Rep_run ρ) m K) = τ) }"
      using Suc_leD not_less_eq_eq by fastforce
    moreover have "{ ρ. hamlet ((Rep_run ρ) n K) = True ∧ time ((Rep_run ρ) n K) = τ
                        ∨ (∃m≥Suc n. hamlet ((Rep_run ρ) m K) = True ∧ time ((Rep_run ρ) m K) = τ) }
                 = ⟦ K ⇑ n ⟧⇩s⇩y⇩m⇩r⇩u⇩n ∩ ⟦ K ⇓ n @ ⌊ τ ⌋ ⟧⇩s⇩y⇩m⇩r⇩u⇩n ∪ ⟦ K sporadic τ ⟧⇩T⇩E⇩S⇩L⇗≥ Suc n⇖"
      by (simp add: Collect_conj_eq Collect_disj_eq)
    ultimately show ?thesis by auto
  qed

lemma TESL_interp_stepwise_timedelayed_coind_unfold:
  shows "⟦ master time-delayed by δτ on measuring implies slave ⟧⇩T⇩E⇩S⇩L⇗≥ n⇖ =
    (⟦ master ¬⇑ n ⟧⇩s⇩y⇩m⇩r⇩u⇩n ∪ ⟦ master ⇑ n ⟧⇩s⇩y⇩m⇩r⇩u⇩n ∩ ⟦ slave sporadic ⌊τ⇩v⇩a⇩r(measuring, n) ⊕ δτ⌋ on measuring ⟧⇩T⇩E⇩S⇩L⇗≥ n⇖)
    ∩ ⟦ master time-delayed by δτ on measuring implies slave ⟧⇩T⇩E⇩S⇩L⇗≥ Suc n⇖"
  proof -
    have "{ ρ. ∀m≥n. hamlet ((Rep_run ρ) m master) ⟶
                 (let measured_time = time ((Rep_run ρ) m measuring) in
                  ∃p ≥ m. hamlet ((Rep_run ρ) p slave) ∧ time ((Rep_run ρ) p measuring) = measured_time + δτ)}
         = { ρ. hamlet ((Rep_run ρ) n master) ⟶
                 (let measured_time = time ((Rep_run ρ) n measuring) in
                  ∃p ≥ n. hamlet ((Rep_run ρ) p slave) ∧ time ((Rep_run ρ) p measuring) = measured_time + δτ)}
           ∩ { ρ. ∀m≥Suc n. hamlet ((Rep_run ρ) m master) ⟶
                 (let measured_time = time ((Rep_run ρ) m measuring) in
                  ∃p ≥ m. hamlet ((Rep_run ρ) p slave) ∧ time ((Rep_run ρ) p measuring) = measured_time + δτ)}"
      by (smt Collect_cong Collect_conj_eq Suc_leD eq_refl le_antisym not_less_eq_eq)
    then show ?thesis by auto
  qed

fun TESL_interpretation_stepwise :: "TESL_formula ⇒ nat ⇒ run set" ("⟦⟦ _ ⟧⟧⇩T⇩E⇩S⇩L⇗≥ _⇖") where
    "⟦⟦ [] ⟧⟧⇩T⇩E⇩S⇩L⇗≥ n⇖ = { _. True }"
  | "⟦⟦ φ # Φ ⟧⟧⇩T⇩E⇩S⇩L⇗≥ n⇖ = ⟦ φ ⟧⇩T⇩E⇩S⇩L⇗≥ n⇖ ∩ ⟦⟦ Φ ⟧⟧⇩T⇩E⇩S⇩L⇗≥ n⇖"

lemma TESL_interpretation_stepwise_fixpoint:
  "⟦⟦ Φ ⟧⟧⇩T⇩E⇩S⇩L⇗≥ n⇖ = ⋂ ((λφ. ⟦ φ ⟧⇩T⇩E⇩S⇩L⇗≥ n⇖) ` set Φ)"
  proof (induct Φ)
    case Nil
    then show ?case by simp
  next
    case (Cons a Φ)
    then show ?case by auto
  qed

  next
